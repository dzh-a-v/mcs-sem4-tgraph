#include "include/Dijkstra.h"
#include <limits>
#include <algorithm>
#include <unordered_map>
#include <vector>

DijkstraAlgorithm::DijkstraAlgorithm(const AdjacencyGraph& graph)
    : m_graph(graph)
{}

DijkstraResult DijkstraAlgorithm::findShortestPaths(int startVertex, std::optional<int> targetVertex) {
    DijkstraResult result;
    result.iterations = 0;
    result.targetDistance = std::numeric_limits<double>::infinity();

    if (!m_graph.hasVertex(startVertex)) {
        return result;
    }

    constexpr double INF = std::numeric_limits<double>::infinity();

    auto vertexIds = m_graph.vertexIds();
    int vertexCount = m_graph.size();

    // Build hash map for O(1) vertex ID to index lookup
    std::unordered_map<int, int> idToIndex;
    for (int i = 0; i < vertexCount; ++i) {
        result.iterations++;
        idToIndex[vertexIds[i]] = i;
    }

    // Initialize distances to infinity
    std::vector<double> dist(vertexCount, INF);
    std::vector<bool> finalized(vertexCount, false);
    std::vector<int> predecessor(vertexCount, -1);

    int startIndex = idToIndex[startVertex];
    dist[startIndex] = 0.0;
    result.distances[startVertex] = 0.0;

    // Classic O(V^2) Dijkstra: iterate V times, each time selecting minimum
    for (int iter = 0; iter < vertexCount; ++iter) {
        result.iterations++;  // Count: outer loop iteration

        // Find minimum distance vertex among unvisited (O(V) scan)
        int minDistIdx = -1;
        double minDist = INF;
        for (int v = 0; v < vertexCount; ++v) {
            result.iterations++;  // Count: finding minimum (O(V) operation)
            if (!finalized[v] && dist[v] < minDist) {
                minDist = dist[v];
                minDistIdx = v;
            }
        }

        // No reachable vertex found
        if (minDistIdx == -1 || dist[minDistIdx] == INF) {
            break;
        }

        finalized[minDistIdx] = true;
        int current = vertexIds[minDistIdx];

        // Store distance in result
        result.distances[current] = dist[minDistIdx];

        // If we reached the target, we can stop
        if (targetVertex.has_value() && current == targetVertex.value()) {
            result.targetDistance = dist[minDistIdx];
            result.shortestPath = reconstructPath(startVertex, current, predecessor);
            return result;
        }

        // Relax edges from current vertex
        for (const auto& [neighbor, weight] : m_graph.neighbors(current)) {
            result.iterations++;  // Count: examining an edge

            auto neighborIt = idToIndex.find(neighbor);
            if (neighborIt == idToIndex.end()) {
                continue;
            }
            int neighborIndex = neighborIt->second;

            if (finalized[neighborIndex]) {
                continue;
            }

            double newDist = dist[minDistIdx] + weight;

            if (newDist < dist[neighborIndex]) {
                dist[neighborIndex] = newDist;
                predecessor[neighborIndex] = minDistIdx;
                result.distances[neighbor] = newDist;
                result.predecessors[neighbor] = current;
            }
        }
    }

    // If target was specified, reconstruct path
    if (targetVertex.has_value()) {
        int target = targetVertex.value();
        auto targetIt = idToIndex.find(target);
        if (targetIt != idToIndex.end() && dist[targetIt->second] != INF) {
            result.targetDistance = dist[targetIt->second];
            result.shortestPath = reconstructPath(startVertex, target, predecessor);
        }
    }

    return result;
}

std::vector<int> DijkstraAlgorithm::reconstructPath(int startVertex, int targetVertex,
                                                     const std::vector<int>& predecessors) const {
    std::vector<int> path;

    if (startVertex == targetVertex) {
        path.push_back(startVertex);
        return path;
    }

    // Backtrack from target to start using index-based predecessors
    auto vertexIds = m_graph.vertexIds();
    int current = targetVertex;
    
    while (true) {
        path.push_back(current);
        if (current == startVertex) break;
        
        auto it = std::find(vertexIds.begin(), vertexIds.end(), current);
        if (it == vertexIds.end()) break;
        
        int idx = std::distance(vertexIds.begin(), it);
        if (predecessors[idx] == -1) break;
        
        current = vertexIds[predecessors[idx]];
    }

    // Reverse to get start -> target order
    std::reverse(path.begin(), path.end());

    // Verify path starts at startVertex
    if (path.empty() || path.front() != startVertex) {
        return {};
    }

    return path;
}
